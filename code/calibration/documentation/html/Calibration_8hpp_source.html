180 void calibrate(Chessboard* chessboards[]) {
181 // Count number of calibration points
182 int num_points = 0;
183 for (int i = 0; chessboards[i] != NULL; i++) {
184 num_points += chessboards[i]->corners.size();
185 }
186
187 // Create and fill Matrix A (Section 4.2 from preparatory research training)
188 Eigen::MatrixXd A(num_points*2, 12);
189 for (int k = 0; chessboards[k] != NULL; k++) {
190 int num_cols = chessboards[k]->horizontal_squares-1;
191 int num_rows = chessboards[k]->vertical_squares-1;
192 double Ly = (double)chessboards[k]->world_coordinates_mm[1];
193 double Lz = (double)chessboards[k]->world_coordinates_mm[2];
194 for (int i = 0; i < num_rows; i++) {
195 double Lx = (double)chessboards[k]->world_coordinates_mm[0]; // FIX THIS
196 for (int j = 0; j < num_cols; j++) {
197 int u = chessboards[k]->corners[i * num_cols + j].x;
198 int v = chessboards[k]->corners[i * num_cols + j].y;
199
200 int index = (k*chessboards[k]->corners.size()) + (i * num_cols + j);
201 A.row(2*index) << Lx, Ly, Lz, 1, 0, 0, 0, 0, -u*Lx, -u*Ly, -u*Lz, -u;
202 A.row(2*index+1) << 0, 0, 0, 0, Lx, Ly, Lz, 1, -v*Lx, -v*Ly, -v*Lz, -v;
203
204 Lx += (double)chessboards[k]->square_size_mm;
205 }
206 Ly -= (double)chessboards[k]->square_size_mm;
207 }
208 }
209
210 // Find Eigenvectors and Eigenvalues (Section 4.2 from preparatory research training)
211 Eigen::MatrixXd ATA = A.transpose() * A;
212 Eigen::EigenSolver<Eigen::MatrixXd> es;
213 Eigen::VectorXd eigenvalues(12, 1);
214 Eigen::MatrixXd eigenvectors(12, 12);
215
216 es.compute(ATA, true);
217 eigenvalues = es.eigenvalues().real();
218 eigenvectors = es.eigenvectors().real();
219
220 // Find Eigenvalue and Eigenvector combination with smallest Loss value (Section 4.2 from preparatory research training)
221 float min = FLT_MAX;
222 uint8_t best_index = 0;
223 for (uint8_t i = 0; i < 12; i++) {
224 float lambda = eigenvalues.row(i).coeff(0, 0);
225 Eigen::MatrixXd x_normalized = eigenvectors.col(i);
226 float loss_func = (x_normalized.transpose() * ATA * x_normalized).coeff(0, 0) - lambda * ((x_normalized.transpose() * x_normalized).coeff(0, 0) - 1);
227 if (loss_func < min) {
228 best_index = i;
229 }
230 }
231
232 // Fill matrix P
233 Eigen::VectorXd p = es.eigenvectors().real().col(best_index);
234 for (uint8_t i = 0; i < 3; i++) {
235 for (uint8_t j = 0; j < 4; j++) {
236 calibration_data.P(i, j) = p(i*4+j);
237 }
238 }
239
240 // Create a new matrix P_1 by omitting the last column of P (section 3.1 from bachelor thesis)
241 Eigen::MatrixXd P_1 = calibration_data.P.leftCols(calibration_data.P.cols() - 1);
242 // Create a new matrix P_2 by extracting the last column of P (section 3.1 from bachelor thesis)
243 Eigen::MatrixXd P_2 = calibration_data.P.col(calibration_data.P.cols() - 1);
244
245 // RQ-Decomposition of P_1 (according to section 2 of bachelor thesis)
246 Eigen::MatrixXd pAT = P_1.colwise().reverse().transpose(); // cfr. Equation 1 of bachelor thesis
247 Eigen::HouseholderQR<Eigen::MatrixXd> qr(pAT); // Perform QR-Decomposition
248 Eigen::MatrixXd q_tilde = qr.householderQ();
249 Eigen::MatrixXd r_tilde = qr.matrixQR().triangularView<Eigen::Upper>();
250
251 // Equation 8 from bachelor thesis
252 Eigen::MatrixXd Q = q_tilde.transpose().colwise().reverse();
253 Eigen::MatrixXd R = r_tilde.transpose().colwise().reverse().rowwise().reverse();
254
255 // fill calibration_data
256 calibration_data.K_inv = R.inverse();
257 calibration_data.R_inv = Q.inverse();
258 calibration_data.t = calibration_data.K_inv * P_2; // see Equation 9 from bachelor thesis
259 }
