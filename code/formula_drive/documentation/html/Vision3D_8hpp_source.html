118 void calculatePosition(std::vector<Cone>* cones) {

120 // Calculate the center of the cone
121 float center_x = 0;
122 float center_y = 0;
123 for(std::pair<float, float> keypoint: cone.keypoints) {
124 center_x += (float)cone.start_x + keypoint.first*cone.width;
125 center_y += (float)cone.start_y + keypoint.second*cone.height;
126 }
127 center_x /= 7;
128 center_y /= 7;
129
130 // Calculate the world coordinates of the cone
131 float z_height = 16.5; // the height of the cone is known
132 Eigen::VectorXd pixel_coord(3); // vector to store the pixel_coordinates of the center of the cone
133 pixel_coord << center_x, center_y, 1; // populate pixel_coord
134 float LzC = (z_height + calibration_data.F(2)) / (calibration_data.E * pixel_coord)(2); // calculate LzC according to Equation 15 of bachelor thesis
135 Eigen::VectorXd Lw = LzC * calibration_data.E * pixel_coord - calibration_data.F; // calculate Lw according to Equation 14 of bachelor thesis
136
137 // Store the world coordinates in the cone object

139 cone.world_coordinates_mm[1] = (int)Lw(1);
