167 std::streamsize size = file.tellg();

171 if (!file.read(buffer.data(), size)) {
172 throw std::runtime_error("Unable to read engine file");

174 file.close();
175
176 // Create a runtime to deserialize the engine file.
177 Logger logger;
178 runtime = std::unique_ptr<nvinfer1::IRuntime> {nvinfer1::createInferRuntime(logger)};
179
180 // Set the device index
181 int ret = cudaSetDevice(device_index);
182 if (ret != 0) {
183 int num_GPUs;
184 cudaGetDeviceCount(&num_GPUs);
185 std::string error_message = "Unable to set GPU device index to: " + std::to_string(device_index) +
186 ". Note, your device has " + std::to_string(num_GPUs) + " CUDA-capable GPU(s).";
187 throw std::runtime_error(error_message);
188 }
189
190 // Create an engine, a representation of the optimized model.
191 engine = std::unique_ptr<nvinfer1::ICudaEngine>(runtime->deserializeCudaEngine(buffer.data(), buffer.size()));
192
193 // The execution context contains all of the state associated with a particular invocation
194 context = std::unique_ptr<nvinfer1::IExecutionContext>(engine->createExecutionContext());
195
196 // Storage for holding the input and output buffers
197 // This will be passed to TensorRT for inference
198 buffers.resize(engine->getNbIOTensors());
199
200 // Create a cuda stream
201 cudaStream_t stream;
202 checkCudaErrorCode(cudaStreamCreate(&stream));
203
204 // Allocate GPU memory for input and output buffers
205 for (int i = 0; i < engine->getNbIOTensors(); i++) {
206 const char* tensor_name = engine->getIOTensorName(i);
207 const nvinfer1::TensorIOMode tensor_type = engine->getTensorIOMode(tensor_name);
208 const nvinfer1::Dims tensor_shape = engine->getTensorShape(tensor_name);
209 int max_number_of_batches = tensor_shape.d[0];
210 if (max_number_of_batches == -1) { // the network allows unlimited batch size
211 max_number_of_batches = max_batch_size;
212 }
213 int number_of_channels = tensor_shape.d[1];
214 // Store information about the input and output dimensions
215 if (tensor_type == nvinfer1::TensorIOMode::kINPUT) { // the binding is an input
216 int input_height = tensor_shape.d[2];
217 int input_width = tensor_shape.d[3];
218 TensorDimensions tensor_dimensions = TensorDimensions(max_number_of_batches, number_of_channels, input_width, input_height);
219 input_dimensions.emplace_back(tensor_dimensions);
220
221 // Allocate memory for the input (allocate enough to fit the max batch size, we could end up using less later)
222 int input_size_bytes = tensor_dimensions.size * sizeof(float);
223 checkCudaErrorCode(cudaMallocAsync(&buffers[i], input_size_bytes, stream));
224 } else if (tensor_type == nvinfer1::TensorIOMode::kOUTPUT) { // The binding is an output
225 int number_of_anchors = tensor_shape.d[2];
226 TensorDimensions tensor_dimensions = TensorDimensions(max_number_of_batches, number_of_channels, number_of_anchors);
227 output_dimensions.emplace_back(tensor_dimensions);
228
229 // Allocate memory for the output
230 int output_size_bytes = tensor_dimensions.size * sizeof(float);
231 checkCudaErrorCode(cudaMallocAsync(&buffers[i], output_size_bytes, stream));
232 } else {
233 throw std::runtime_error("Error, IO Tensor is neither an input or output!");
234 }
235 }
236 // Synchronize and destroy the cuda stream
237 checkCudaErrorCode(cudaStreamSynchronize(stream));
