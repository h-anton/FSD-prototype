62bool intersects(Point<T1>& p1, Point<T1>& p2, Point<T2>& p3, Point<T2>& p4) {
63 // Calculate the orientation of triplet (p1, p2, p3)
64 auto orientation1 = (p2.getY() - p1.getY()) * (p3.getX() - p2.getX()) -
65 (p3.getY() - p2.getY()) * (p2.getX() - p1.getX());
66
67 // Calculate the orientation of triplet (p1, p2, p4)
68 auto orientation2 = (p2.getY() - p1.getY()) * (p4.getX() - p2.getX()) -
69 (p4.getY() - p2.getY()) * (p2.getX() - p1.getX());
70
71 // Calculate the orientation of triplet (p3, p4, p1)
72 auto orientation3 = (p4.getY() - p3.getY()) * (p1.getX() - p4.getX()) -
73 (p1.getY() - p4.getY()) * (p4.getX() - p3.getX());
74
75 // Calculate the orientation of triplet (p3, p4, p2)
76 auto orientation4 = (p4.getY() - p3.getY()) * (p2.getX() - p4.getX()) -
77 (p2.getY() - p4.getY()) * (p4.getX() - p3.getX());
78
79 // Check if orientations are different
80 return (orientation1 * orientation2 < 0) && (orientation3 * orientation4 < 0);

302 void findPath() {
303 // Vectors for the (un)sorted path_points. The first point of the path is always (0, 0)
304 std::vector<Point<int>> path_points_unsorted;
305 std::vector<Point<int>> path_points_sorted;
306 path_points_sorted.push_back(Point(0, 0));
307 // This set is used to avoid duplicates in path_points_unsorted
308 std::set<Point<int>> unique_points;
309
310 // Find all the points in the middle of the track
311 for (Edge<double> edge: edges) {
312 // A point on the middle of the track always lies on an edge connecting 2 points of different colors
313 if (!edge.same_color) {
314 // Find the middle of that edge
315 double x = (edge.point1.getX()+edge.point2.getX())/2;
316 double y = (edge.point1.getY()+edge.point2.getY())/2;
317 Point<int> point((int)x, (int)y);
318
319 // Check if the point is already in the set, in order to avoid duplicates
320 if (unique_points.find(point) == unique_points.end()) {
321 // If not found, add it to both the vector and the set
322 path_points_unsorted.push_back(point);
323 unique_points.insert(point);
324 }
325 }
326 }
327 int number_of_points = path_points_unsorted.size();
328
329 // sort these points in order of appearance
330 while (path_points_sorted.size() < number_of_points + 1) {
331 Point<int> last_sorted_point = path_points_sorted.back();
332 // Find the point which is closest to last_sorted_point
333 std::vector<Point<int>>::iterator iterator_next_point = std::min_element(path_points_unsorted.begin(), path_points_unsorted.end(),
334 [last_sorted_point](const Point<int>& p1, const Point<int>& p2) -> bool {
335 return compareDistance(p1, p2, last_sorted_point);
336 });
337 Point<int> next_point = *iterator_next_point;
338
339 // If the edge between last_sorted_point and next_point intersects the track boundaries, we have reached the end of the path
340 bool intersects_track_edge = false;
341 for (Edge<double> edge: edges) {
342 // An edge is a track boundary if it connects 2 points of the same color
343 if (edge.same_color) {
344 Point<double> boundary1 = edge.point1;
345 Point<double> boundary2 = edge.point2;
346 if (intersects(next_point, last_sorted_point, boundary1, boundary2)) {
347 intersects_track_edge = true;
348 break;
349 }
350 }
351 }
352 if (intersects_track_edge) {
353 break;
354 }
355
356 path_points_sorted.push_back(next_point);
357 path_points_unsorted.erase(iterator_next_point);
358 }
359
360 path = path_points_sorted;
361 }
