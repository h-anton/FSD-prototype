144 std::vector<float> output_vector = result[0][0];

158 for (int i = 0; i < num_anchors; i++) {
159 float* output_ptr = output.row(i).ptr<float>();
160 // the model gives a score to each possible 'class'
161 //float* max_score_ptr = std::max_element(output_ptr+4, output_ptr+9);
162 float* ptr_score_first_class = &output.at<float>(i, 4);
163 float* ptr_score_last_class = &output.at<float>(i, 9);

167 if (score > probability_threshold) {

171 float h = output.at<float>(i, 3);

182 int start_y_full_frame = (int)(start_y / input_dimensions[0].height * frame.rows);

188 start_y_full_frame = std::max(0, start_y_full_frame);
189 width_full_frame = std::min(frame.cols-start_x_full_frame, width_full_frame);
190 height_full_frame = std::min(frame.rows-start_y_full_frame, height_full_frame);
191
192 // Discard bounding boxes which are too big to be a cone
193 // Discard bounding boxes with abnormal shapes
194 if (width_full_frame * height_full_frame < 0.015 * frame.cols * frame.rows

227 cv::cuda::GpuMat blobFromGpuMats(const std::vector<cv::cuda::GpuMat>& batchInput, bool normalize) {
228 cv::cuda::GpuMat gpu_dst(1, batchInput[0].rows * batchInput[0].cols * batchInput.size(), CV_8UC3);
229
230 size_t width = batchInput[0].cols * batchInput[0].rows;
231 for (size_t img = 0; img < batchInput.size(); img++) {
232 std::vector<cv::cuda::GpuMat> input_channels{
233 cv::cuda::GpuMat(batchInput[0].rows, batchInput[0].cols, CV_8U, &(gpu_dst.ptr()[0 + width * 3 * img])),
234 cv::cuda::GpuMat(batchInput[0].rows, batchInput[0].cols, CV_8U, &(gpu_dst.ptr()[width + width * 3 * img])),
235 cv::cuda::GpuMat(batchInput[0].rows, batchInput[0].cols, CV_8U,
236 &(gpu_dst.ptr()[width * 2 + width * 3 * img]))
237 };
238 cv::cuda::split(batchInput[img], input_channels); // HWC -> CHW
239 }
240
241 cv::cuda::GpuMat m_float;
242 if (normalize) {
243 // [0.f, 1.f]
244 gpu_dst.convertTo(m_float, CV_32FC3, 1.f / 255.f);
245 } else {
246 // [0.f, 255.f]
247 gpu_dst.convertTo(m_float, CV_32FC3);
248 }
249
250 // Apply scaling and mean subtraction
251 std::array<float, 3> sub_values{0.f, 0.f, 0.f};
252 std::array<float, 3> div_values{1.f, 1.f, 1.f};
253 cv::cuda::subtract(m_float, cv::Scalar(sub_values[0], sub_values[1], sub_values[2]), m_float, cv::noArray(), -1);
254 cv::cuda::divide(m_float, cv::Scalar(div_values[0], div_values[1], div_values[2]), m_float, 1, -1);
255
256 return m_float;
257 }
